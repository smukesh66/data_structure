#include<stdio.h>

int main(){
int a[2][2],b[2][2],sum[2][2]={0},i,j;
printf("enter the matrix a");
for(i=0;i<2;i++){
for(j=0;j<2;j++)
scanf("%d",&a[i][j]);
}
printf("enter the matrix b");
for(i=0;i<2;i++)
{
for(j=0;j<2;j++)
{
scanf("%d",&b[i][j]);
}
}

printf("addition of matrix");
for(i=0;i<2;i++){
for(j=0;j<2;j++)
sum[i][j]= a[i][j] + b[i][j];
}
printf("the result");
for(i=0;i<2;i++){
for(j=0;j<2;j++)
printf("%d",sum[i][j]);
}
return 0;
}
